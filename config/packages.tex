\DeclareDelimFormat{nameyeardelim}{\addcomma\space}

% Hilangkan bulan/tanggal akses
\AtEveryBibitem{%
  \clearfield{month}
  \clearfield{day}
  \clearfield{urldate}
  \clearfield{langid}
  \clearlist{language}
}

% Nama penulis
\DeclareNameAlias{author}{family-given}

% Pemisah penulis
\DeclareDelimFormat{nameyeardelim}{\addcomma\space}
\DeclareDelimFormat{finalnamedelim}{\addspace dan\space}

% Antar banyak sitasi:
% 2 sitasi  -> A dan B
% 3+ sitasi -> A, B dan C
\DeclareDelimFormat{multicitedelim}{%
  \ifnumless{\value{multicitecount}}{\value{multicitetotal}}
    {\addcomma\space}
    {\addspace dan\space}%
}

% Sitasi > 2 penulis menjadi dkk.
\ExecuteBibliographyOptions{
  maxcitenames=2,
  mincitenames=1,
  uniquelist=false,
  uniquename=false
}

% Tahun pakai kurung
\renewbibmacro*{date+extradate}{%
  \printtext[parens]{%
    \printfield{year}%
    \printfield{extradate}}}

% Judul artikel tanpa kutip
\DeclareFieldFormat[article]{title}{#1}

% Format artikel jurnal: Penulis (Tahun) : Judul, Jurnal, Volume, Halaman.
\DeclareBibliographyDriver{article}{%
  \printnames{author}%
  \space
  \printtext[parens]{\printfield{year}}%
  \space:\space
  \printfield{title}%
  \addcomma\space
  \printfield{journaltitle}%
  \addcomma\space
  \printfield{volume}%
  \addcomma\space
  \printfield{pages}%
  \addperiod
  \finentry
}

% Nama jurnal italic
\DeclareFieldFormat[article]{journaltitle}{\mkbibemph{#1}}

% Volume bold
\DeclareFieldFormat[article]{volume}{\textbf{#1}}

% Nomor halaman
\DeclareFieldFormat{pages}{#1}

% Hilangkan "In:"
\renewbibmacro{in:}{}

% Style dari website
\DeclareFieldFormat{rawurl}{\href{#1}{#1}}

\DeclareBibliographyDriver{online}{%
  \printfield{title}%
  \addcomma\space
  diperoleh melalui situs internet:\space
  \printfield[rawurl]{url}%
  \addperiod\space
  Diunduh pada tanggal\space
  \printurldate
  \finentry
}

\DeclareFieldFormat{urldate}{%
  \thefield{urlday}~%
  \ifcase\thefield{urlmonth}\or Januari\or Februari\or Maret\or April\or Mei\or Juni\or Juli\or Agustus\or September\or Oktober\or November\or Desember\fi
  \space\thefield{urlyear}%
}
